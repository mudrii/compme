//! P2: warm-up, then measure first-token + short-completion latency. Greedy decode.
use std::num::NonZeroU32;
use std::path::PathBuf;
use std::time::Instant;
use llama_cpp_2::llama_backend::LlamaBackend;
use llama_cpp_2::model::{AddBos, LlamaModel};
use llama_cpp_2::model::params::LlamaModelParams;
use llama_cpp_2::context::params::LlamaContextParams;
use llama_cpp_2::llama_batch::LlamaBatch;
use llama_cpp_2::sampling::LlamaSampler;

fn run(model_path: &str, prompt: &str, n_predict: usize) -> Result<(), Box<dyn std::error::Error>> {
    let backend = LlamaBackend::init()?;
    let mparams = LlamaModelParams::default().with_n_gpu_layers(999);
    let model = LlamaModel::load_from_file(&backend, PathBuf::from(model_path), &mparams)?;
    let cparams = LlamaContextParams::default().with_n_ctx(Some(NonZeroU32::new(2048).unwrap()));

    // helper: one short greedy completion, returns (first_token_ms, total_ms, text)
    let mut once = |warm: bool| -> Result<(u128, u128, String), Box<dyn std::error::Error>> {
        let mut ctx = model.new_context(&backend, cparams.clone())?;
        let tokens = model.str_to_token(prompt, AddBos::Always)?;
        let mut batch = LlamaBatch::new(512, 1);
        let last = tokens.len() - 1;
        for (i, t) in tokens.iter().enumerate() {
            batch.add(*t, i as i32, &[0], i == last)?;
        }
        let t0 = Instant::now();
        ctx.decode(&mut batch)?;
        let mut sampler = LlamaSampler::greedy();
        let mut out = String::new();
        let mut decoder = encoding_rs::UTF_8.new_decoder();
        let mut first_ms = 0u128;
        let mut n_cur = batch.n_tokens();
        for n in 0..n_predict {
            let tok = sampler.sample(&ctx, batch.n_tokens() - 1);
            if n == 0 { first_ms = t0.elapsed().as_millis(); }
            if model.is_eog_token(tok) { break; }
            out.push_str(&model.token_to_piece(tok, &mut decoder, true, None)?);
            batch.clear();
            batch.add(tok, n_cur, &[0], true)?;
            n_cur += 1;
            ctx.decode(&mut batch)?;
        }
        let total_ms = t0.elapsed().as_millis();
        let _ = warm;
        Ok((first_ms, total_ms, out))
    };

    let _ = once(true)?;                // warm-up (pays Metal shader compile)
    let (first, total, text) = once(false)?;
    println!("[{}] first_token={}ms total({} tok)={}ms => \"{}\"",
        model_path, first, n_predict, total, text.replace('\n', "\\n"));
    Ok(())
}

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let prompt = "The quick brown fox";
    run("models/qwen2.5-0.5b-instruct-q4_k_m.gguf", prompt, 12)?;
    if std::path::Path::new("models/qwen2.5-0.5b-q4_k_m.gguf").exists() {
        run("models/qwen2.5-0.5b-q4_k_m.gguf", prompt, 12)?;
    }
    Ok(())
}
```

//! P3: read focused field value + left-of-caret context via Accessibility.
//! Run, then focus a TextEdit document and type — it reads every 1s.
use std::os::raw::c_void;
use std::{thread, time::Duration};
use accessibility_sys::{
    AXIsProcessTrusted, AXUIElementCreateSystemWide, AXUIElementCopyAttributeValue,
    AXValueGetValue, AXUIElementRef, AXValueRef, kAXErrorSuccess,
    kAXFocusedUIElementAttribute, kAXValueAttribute, kAXSelectedTextRangeAttribute,
    kAXValueTypeCFRange,
};
use core_foundation::base::{CFType, CFTypeRef, TCFType, CFRange};
use core_foundation::string::{CFString, CFStringRef};

/// Copy an AX attribute as a managed CFType (released on drop). None on error/missing.
unsafe fn copy_attr(el: AXUIElementRef, attr: &str) -> Option<CFType> {
    let cf = CFString::new(attr);
    let mut out: CFTypeRef = std::ptr::null();
    let err = AXUIElementCopyAttributeValue(el, cf.as_concrete_TypeRef(), &mut out);
    if err == kAXErrorSuccess && !out.is_null() {
        Some(CFType::wrap_under_create_rule(out))
    } else {
        None
    }
}

fn main() {
    unsafe {
        if AXIsProcessTrusted() == 0 {
            eprintln!("NOT trusted: grant Accessibility to this terminal, then relaunch it.");
            return;
        }
        let sys = AXUIElementCreateSystemWide();
        println!("Focus a TextEdit doc and type; reading every 1s. Ctrl-C to stop.");
        loop {
            let Some(focused) = copy_attr(sys, kAXFocusedUIElementAttribute) else {
                println!("(no focused element)");
                thread::sleep(Duration::from_secs(1));
                continue;
            };
            let el = focused.as_CFTypeRef() as AXUIElementRef;

            let value = copy_attr(el, kAXValueAttribute).map(|v| {
                CFString::wrap_under_get_rule(v.as_CFTypeRef() as CFStringRef).to_string()
            });

            let mut caret: isize = 0;
            if let Some(r) = copy_attr(el, kAXSelectedTextRangeAttribute) {
                let mut range = CFRange { location: 0, length: 0 };
                if AXValueGetValue(r.as_CFTypeRef() as AXValueRef, kAXValueTypeCFRange,
                    &mut range as *mut _ as *mut c_void) != 0 {
                    caret = range.location;
                }
            }

            match value {
                Some(v) => {
                    let left: String = v.chars().take(caret.max(0) as usize).collect();
                    let tail: String = {
                        let mut t: Vec<char> = left.chars().rev().take(40).collect();
                        t.reverse();
                        t.into_iter().collect()
                    };
                    println!("caret={} left_tail=\"{}\"", caret, tail.replace('\n', "\\n"));
                }
                None => println!("(focused element exposes no AXValue — unsupported field)"),
            }
            thread::sleep(Duration::from_secs(1));
        }
    }
}
```

//! P4: caret-rect ladder — tier1 (zero-length at caret) then tier3 (prev-char right edge).
//! Run, focus TextEdit, type; prints the caret screen rect + which tier produced it.
use std::os::raw::c_void;
use std::{thread, time::Duration};
use accessibility_sys::{
    AXIsProcessTrusted, AXUIElementCreateSystemWide, AXUIElementCopyAttributeValue,
    AXUIElementCopyParameterizedAttributeValue, AXValueCreate, AXValueGetValue,
    AXUIElementRef, AXValueRef, kAXErrorSuccess,
    kAXFocusedUIElementAttribute, kAXSelectedTextRangeAttribute,
    kAXBoundsForRangeParameterizedAttribute, kAXValueTypeCFRange, kAXValueTypeCGRect,
};
use core_foundation::base::{CFType, CFTypeRef, TCFType, CFRange};
use core_foundation::string::CFString;
use core_graphics::geometry::{CGRect, CGPoint, CGSize};

unsafe fn copy_attr(el: AXUIElementRef, attr: &str) -> Option<CFType> {
    let cf = CFString::new(attr);
    let mut out: CFTypeRef = std::ptr::null();
    let err = AXUIElementCopyAttributeValue(el, cf.as_concrete_TypeRef(), &mut out);
    if err == kAXErrorSuccess && !out.is_null() { Some(CFType::wrap_under_create_rule(out)) } else { None }
}

unsafe fn bounds_for_range(el: AXUIElementRef, location: isize, length: isize) -> Option<CGRect> {
    let range = CFRange { location, length };
    let axval = AXValueCreate(kAXValueTypeCFRange, &range as *const _ as *const c_void);
    if axval.is_null() { return None; }
    let _hold = CFType::wrap_under_create_rule(axval as CFTypeRef); // release on drop
    let pname = CFString::new(kAXBoundsForRangeParameterizedAttribute);
    let mut out: CFTypeRef = std::ptr::null();
    let err = AXUIElementCopyParameterizedAttributeValue(
        el, pname.as_concrete_TypeRef(), axval as CFTypeRef, &mut out);
    if err != kAXErrorSuccess || out.is_null() { return None; }
    let held = CFType::wrap_under_create_rule(out);
    let mut rect = CGRect { origin: CGPoint { x: 0.0, y: 0.0 }, size: CGSize { width: 0.0, height: 0.0 } };
    if AXValueGetValue(held.as_CFTypeRef() as AXValueRef, kAXValueTypeCGRect,
        &mut rect as *mut _ as *mut c_void) != 0 {
        Some(rect)
    } else { None }
}

/// Reject empty + container-sized rects (Electron returns the whole composer).
fn usable(r: &CGRect) -> bool { r.size.height > 0.0 && r.size.height < 200.0 && r.size.width < 2000.0 }

fn main() {
    unsafe {
        if AXIsProcessTrusted() == 0 { eprintln!("grant Accessibility, relaunch."); return; }
        let sys = AXUIElementCreateSystemWide();
        println!("Focus TextEdit, type; printing caret rect every 1s. Ctrl-C to stop.");
        loop {
            if let Some(focused) = copy_attr(sys, kAXFocusedUIElementAttribute) {
                let el = focused.as_CFTypeRef() as AXUIElementRef;
                let mut caret: isize = 0;
                if let Some(r) = copy_attr(el, kAXSelectedTextRangeAttribute) {
                    let mut range = CFRange { location: 0, length: 0 };
                    if AXValueGetValue(r.as_CFTypeRef() as AXValueRef, kAXValueTypeCFRange,
                        &mut range as *mut _ as *mut c_void) != 0 { caret = range.location; }
                }
                let mut tier = "none";
                let mut rect: Option<CGRect> = None;
                if let Some(r) = bounds_for_range(el, caret, 0) { if usable(&r) { tier = "exact"; rect = Some(r); } }
                if rect.is_none() && caret > 0 {
                    if let Some(r) = bounds_for_range(el, caret - 1, 1) { if usable(&r) { tier = "derived"; rect = Some(r); } }
                }
                match rect {
                    Some(r) => println!("caret={} tier={} rect=({:.0},{:.0} {:.0}x{:.0})",
                        caret, tier, r.origin.x, r.origin.y, r.size.width, r.size.height),
                    None => println!("caret={} tier=none (no usable caret rect -> popup fallback)", caret),
                }
            } else { println!("(no focus)"); }
            thread::sleep(Duration::from_secs(1));
        }
    }
}
```

//! P5: consuming CGEventTap that SWALLOWS F8 (keycode 100), passes all other keys.
//! Proves global interception + that an instant callback doesn't stall other apps.
use core_foundation::runloop::{CFRunLoop, kCFRunLoopCommonModes};
use core_graphics::event::{
    CGEventTap, CGEventTapLocation, CGEventTapPlacement, CGEventTapOptions,
    CGEventType, CGEventField,
};

const KEYCODE_F8: i64 = 100;

fn main() {
    let tap = CGEventTap::new(
        CGEventTapLocation::Session,
        CGEventTapPlacement::HeadInsertEventTap,
        CGEventTapOptions::Default, // active/consuming
        vec![CGEventType::KeyDown],
        |_proxy, _etype, event| {
            // Callback MUST stay non-blocking — slow callbacks get the tap auto-disabled
            // and stall global input. Decide from pre-computed state only.
            let keycode = event.get_integer_value_field(CGEventField::KEYBOARD_EVENT_KEYCODE);
            if keycode == KEYCODE_F8 {
                println!("F8 -> SWALLOWED");
                None
            } else {
                println!("key {keycode} -> passed");
                Some(event.to_owned())
            }
        },
    ).expect("tap create failed — grant Input Monitoring + Accessibility, relaunch terminal");

    let current = CFRunLoop::get_current();
    let source = tap.mach_port.create_runloop_source(0).expect("runloop source");
    unsafe { current.add_source(&source, kCFRunLoopCommonModes); }
    tap.enable();
    println!("Tap active. F8 = swallowed, others = passed. Type in ANOTHER app to confirm no stall. Ctrl-C to stop.");
    CFRunLoop::run_current();
}
```

//! P7: integration smoke. After 3s (focus a TextEdit doc with some text!), read left
//! context + caret rect, generate a short completion locally, show it as grey ghost
//! text at the caret. One-shot. Ctrl-C to quit.
use std::num::NonZeroU32;
use std::os::raw::c_void;
use std::path::PathBuf;
use std::{thread, time::Duration};

use accessibility_sys::{
    AXIsProcessTrusted, AXUIElementCreateSystemWide, AXUIElementCopyAttributeValue,
    AXUIElementCopyParameterizedAttributeValue, AXValueCreate, AXValueGetValue,
    AXUIElementRef, AXValueRef, kAXErrorSuccess, kAXFocusedUIElementAttribute,
    kAXValueAttribute, kAXSelectedTextRangeAttribute, kAXBoundsForRangeParameterizedAttribute,
    kAXValueTypeCFRange, kAXValueTypeCGRect,
};
use core_foundation::base::{CFType, CFTypeRef, TCFType, CFRange};
use core_foundation::string::{CFString, CFStringRef};
use core_graphics::geometry::{CGRect, CGPoint, CGSize};

use llama_cpp_2::llama_backend::LlamaBackend;
use llama_cpp_2::model::{AddBos, LlamaModel, Special};
use llama_cpp_2::model::params::LlamaModelParams;
use llama_cpp_2::context::params::LlamaContextParams;
use llama_cpp_2::llama_batch::LlamaBatch;
use llama_cpp_2::sampling::LlamaSampler;

use objc2::rc::Retained;
use objc2::MainThreadMarker;
use objc2_foundation::{NSRect, NSPoint, NSSize, NSString};
use objc2_app_kit::{
    NSApplication, NSApplicationActivationPolicy, NSPanel, NSWindowStyleMask,
    NSBackingStoreType, NSColor, NSTextField, NSScreen,
};

unsafe fn copy_attr(el: AXUIElementRef, attr: &str) -> Option<CFType> {
    let cf = CFString::new(attr);
    let mut out: CFTypeRef = std::ptr::null();
    if AXUIElementCopyAttributeValue(el, cf.as_concrete_TypeRef(), &mut out) == kAXErrorSuccess && !out.is_null() {
        Some(CFType::wrap_under_create_rule(out))
    } else { None }
}

unsafe fn caret_rect(el: AXUIElementRef, caret: isize) -> Option<CGRect> {
    for (loc, len) in [(caret, 0isize), ((caret - 1).max(0), 1)] {
        let range = CFRange { location: loc, length: len };
        let axval = AXValueCreate(kAXValueTypeCFRange, &range as *const _ as *const c_void);
        if axval.is_null() { continue; }
        let _hold = CFType::wrap_under_create_rule(axval as CFTypeRef);
        let pname = CFString::new(kAXBoundsForRangeParameterizedAttribute);
        let mut out: CFTypeRef = std::ptr::null();
        if AXUIElementCopyParameterizedAttributeValue(el, pname.as_concrete_TypeRef(), axval as CFTypeRef, &mut out) == kAXErrorSuccess && !out.is_null() {
            let held = CFType::wrap_under_create_rule(out);
            let mut r = CGRect { origin: CGPoint { x: 0.0, y: 0.0 }, size: CGSize { width: 0.0, height: 0.0 } };
            if AXValueGetValue(held.as_CFTypeRef() as AXValueRef, kAXValueTypeCGRect, &mut r as *mut _ as *mut c_void) != 0
                && r.size.height > 0.0 && r.size.height < 200.0 {
                return Some(r);
            }
        }
    }
    None
}

fn complete(prompt: &str) -> String {
    let backend = LlamaBackend::init().unwrap();
    let mparams = LlamaModelParams::default().with_n_gpu_layers(999);
    let model = LlamaModel::load_from_file(
        &backend, PathBuf::from("models/qwen2.5-0.5b-instruct-q4_k_m.gguf"), &mparams).unwrap();
    let cparams = LlamaContextParams::default().with_n_ctx(Some(NonZeroU32::new(2048).unwrap()));
    let mut ctx = model.new_context(&backend, cparams).unwrap();
    let tokens = model.str_to_token(prompt, AddBos::Always).unwrap();
    let mut batch = LlamaBatch::new(512, 1);
    let last = tokens.len() - 1;
    for (i, t) in tokens.iter().enumerate() { batch.add(*t, i as i32, &[0], i == last).unwrap(); }
    ctx.decode(&mut batch).unwrap();
    let mut sampler = LlamaSampler::greedy();
    let mut out = String::new();
    let mut decoder = encoding_rs::UTF_8.new_decoder();
    let mut n_cur = batch.n_tokens();
    for _ in 0..8 {
        let tok = sampler.sample(&ctx, batch.n_tokens() - 1);
        if model.is_eog_token(tok) { break; }
        out.push_str(&model.token_to_piece(tok, &mut decoder, true, None).unwrap_or_default());
        batch.clear();
        batch.add(tok, n_cur, &[0], true).unwrap();
        n_cur += 1;
        ctx.decode(&mut batch).unwrap();
    }
    out
}

fn main() {
    unsafe {
        if AXIsProcessTrusted() == 0 { eprintln!("grant Accessibility + Input Monitoring, relaunch."); return; }
    }
    let mtm = MainThreadMarker::new().expect("main thread");
    let app = NSApplication::sharedApplication(mtm);
    app.setActivationPolicy(NSApplicationActivationPolicy::Accessory);

    println!("Focus a TextEdit doc with some text. Reading in 3s...");
    thread::sleep(Duration::from_secs(3));

    let (left, rect) = unsafe {
        let sys = AXUIElementCreateSystemWide();
        match copy_attr(sys, kAXFocusedUIElementAttribute) {
            Some(f) => {
                let el = f.as_CFTypeRef() as AXUIElementRef;
                let value = copy_attr(el, kAXValueAttribute)
                    .map(|v| CFString::wrap_under_get_rule(v.as_CFTypeRef() as CFStringRef).to_string())
                    .unwrap_or_default();
                let mut caret: isize = value.chars().count() as isize;
                if let Some(r) = copy_attr(el, kAXSelectedTextRangeAttribute) {
                    let mut range = CFRange { location: 0, length: 0 };
                    if AXValueGetValue(r.as_CFTypeRef() as AXValueRef, kAXValueTypeCFRange, &mut range as *mut _ as *mut c_void) != 0 { caret = range.location; }
                }
                let left: String = value.chars().take(caret.max(0) as usize).collect();
                (left, caret_rect(el, caret))
            }
            None => (String::new(), None),
        }
    };

    if left.trim().is_empty() { eprintln!("no left context — focus a text field WITH text and retry."); return; }
    println!("left context: \"{}\"", left.replace('\n', "\\n"));
    let completion = complete(left.trim_end());
    println!("completion: \"{}\"", completion.replace('\n', "\\n"));

    let (px, py, ph) = match rect {
        Some(r) => {
            let screen_h = NSScreen::mainScreen(mtm).map(|s| s.frame().size.height).unwrap_or(900.0);
            (r.origin.x, screen_h - r.origin.y - r.size.height, r.size.height.max(18.0))
        }
        None => { println!("(no caret rect — fixed point)"); (400.0, 400.0, 22.0) }
    };

    let frame = NSRect::new(NSPoint::new(px, py), NSSize::new(260.0, ph + 6.0));
    let style = NSWindowStyleMask::Borderless | NSWindowStyleMask::NonactivatingPanel;
    let panel: Retained<NSPanel> = unsafe {
        NSPanel::initWithContentRect_styleMask_backing_defer(
            NSPanel::alloc(mtm), frame, style, NSBackingStoreType::Buffered, false)
    };
    unsafe {
        panel.setOpaque(false);
        panel.setBackgroundColor(Some(&NSColor::clearColor()));
        panel.setLevel(101);
        panel.setIgnoresMouseEvents(true);
        panel.setHidesOnDeactivate(false);
        let label = NSTextField::labelWithString(&NSString::from_str(&completion), mtm);
        label.setFrame(NSRect::new(NSPoint::new(2.0, 2.0), NSSize::new(256.0, ph)));
        label.setTextColor(Some(&NSColor::colorWithWhite_alpha(0.5, 0.9)));
        label.setDrawsBackground(false);
        label.setBezeled(false);
        label.setEditable(false);
        if let Some(c) = panel.contentView() { c.addSubview(&label); }
        panel.orderFrontRegardless();
    }
    println!("Ghost completion shown at caret. Ctrl-C to quit.");
    app.run();
}
